export const generateDescricao = (nome = '') => {
  const s = String(nome).toLowerCase().trim();
  const descricao = ACENTOS[s]
    ? ACENTOS[s]
    : s.split(' ').map(w => w ? w.charAt(0).toUpperCase() + w.slice(1) : '').join(' ');
  return descricao.replace(/[()]/g, '').trim();
};
